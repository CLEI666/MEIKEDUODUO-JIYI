// 更新页面内容
function updateContent(lang) {
    const content = langContent[lang];
    const aboutZh = document.querySelector('.text-about-zh');
    const aboutEn = document.querySelector('.text-about-en');
    const cLeft = document.querySelector('#c_left');
    const bottomSpan = document.querySelector('#bottom span');
    const chatBtn = document.querySelector('.live-chat');
    
    if (aboutZh) aboutZh.innerHTML = content.aboutZh;
    if (aboutEn) {
        aboutEn.innerHTML = content.aboutEn;
        aboutEn.style.display = content.showAboutEn ? 'block' : 'none';
    }
    if (cLeft) cLeft.innerHTML = content.leftContent;
    if (bottomSpan) bottomSpan.innerHTML = content.bottomText;
    if (chatBtn) chatBtn.innerHTML = content.chatBtn;
}

// 切换语言
function toggleLanguage() {
    if (currentLang === 'zh') {
		currentLang = 'en';
	} else if (currentLang === 'en') {
		currentLang = 'ja';
	} else {
		currentLang = 'zh';
	}
    updateContent(currentLang);
}

// 页面加载时初始化
window.addEventListener('DOMContentLoaded', function() {
    updateContent(currentLang);    
    // 绑定模态框关闭事件
    document.getElementById('modalOverlay').addEventListener('click', closeChat);
});

// 打开聊天窗口
function openChat(event) {
    if (event) {
        event.preventDefault();
        event.stopPropagation();
    }
    
    document.getElementById('modalOverlay').classList.add('modal-show');
    document.getElementById('modalContent').classList.add('modal-show');
    
    const form = document.createElement('form');
    form.style.display = 'none';
    form.method = 'POST';
    form.target = 'chatFrame';
    form.action = 'https://chat.net7.net/?' + DOMAIN_REF + '&ln=' + currentLang;
    
    const input = document.createElement('input');
    input.type = 'hidden';
    input.name = 'siteName';
    input.value = DOMAIN_REF;
    form.appendChild(input);
    
    document.body.appendChild(form);
    form.submit();
    setTimeout(function() {
        document.body.removeChild(form);
    }, 100);
    
    document.body.style.overflow = 'hidden';
    
    return false;
}

// 关闭聊天窗口
function closeChat() {
    document.getElementById('modalOverlay').classList.remove('modal-show');
    document.getElementById('modalContent').classList.remove('modal-show');
    document.body.style.overflow = '';
    document.getElementById('chatFrame').src = 'about:blank';
}

// 监听消息关闭聊天
window.addEventListener('message', function(event) {
    if (event.data === 'closeChat') {
        closeChat();
    }
});

// ESC键关闭聊天
document.addEventListener('keydown', function(event) {
    if (event.key === 'Escape') {
        closeChat();
    }
});
